- 基础01背包(一维滚动数组)
- 适用场景: 每种物品只有一件，选或不选


    //每一轮保留 -> 考虑了前i个物品的情况下， 容量是j时的最大价值

```C++
const int N = 1010; // 物品最大数量
const int M = 1010; // 背包容量最大数量

int n, m;           // 物品个数、背包容量 
int v[N], w[N];     // v[i]体积、w[i]价值
int f[M]            // f[j]容量为j时的最大价值

int main(){
    std::cin >> n >> m;
    for(int i = 0; i < n; ++i){
        std::cin >> v[i] >> w[i];
    }

    for(int i = 0; i < n; ++i){
        for(int j = m; j >= v[i]; j--){            // 依次考虑每个物品，决定要不要把它放进背包
            f[j] = max(f[j], f[j - v[i]] + w[i]);  // 考虑了i-1个物品后，考虑背包容量为j时，要不要放入这个物品。而放不放入取决于不选这个物品时的最大价值和在上一轮基础上选了这个物品的情况哪个价值更高
        }
    }


    std::cout << f[m] << std::endl;
}
```
